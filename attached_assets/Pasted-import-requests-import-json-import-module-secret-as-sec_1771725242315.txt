import requests
import json
import module.secret as secret

plaid_env = secret.plaid_env()

def env_host():
  if plaid_env == "prod":
    return "https://production.plaid.com"

  elif plaid_env == "dev":
    return "https://development.plaid.com"

  else:
    return "https://sandbox.plaid.com"

def transactions_sync(access_token, cursor):
  # Plaid Transactions Sync Request endpoint
  url = env_host() + "/transactions/sync"

  # Header for authorization
  headers = {"Content-Type": "application/json"}
 
  # Data to be sent in the POST request
  data = {
      "client_id": secret.plaid_client_id,
      "secret": secret.plaid_key,
      "access_token": access_token,
      "cursor": cursor,
      "count": 250,  # Optional, defaults to 250. Maximum: 500
  }

  # Send POST request
  response = requests.post(url, headers=headers, json=data)

  # Check for successful response
  if response.status_code == 200:
      # Parse JSON response
      response_data = json.loads(response.text)
      print("Transactions sync request successful!")
      return response_data
      # Access data from response_data dictionary
  else:
      print(f"Request failed with status code: {response.status_code}")
      print(response.text)
      return None

def create_link_token(user_id):
  url = env_host() + "/link/token/create"
  headers = {"Content-Type": "application/json"}
  data = {
    "client_id": secret.plaid_client_id,
    "secret": secret.plaid_key,
    "user": {
      "client_user_id": user_id
    },
    "client_name": "Cash Cushion",
    "products": ["transactions"],
    "transactions": {
      "days_requested": 730
    },
    "country_codes": ["US"],
    "language": "en",
    "webhook": "https://plaid.webhook.cashcushion.net",
    "redirect_uri": "https://www.cashcushion.net/app/link", # For oauth connections in prod
    "update": { "account_selection_enabled": True },
    "account_filters": {
      "depository": {
        "account_subtypes": ["checking", "savings"]
      },
    }
  }
  response = requests.post(url, headers=headers, json=data)
  if response.status_code == 200:
    response_data = json.loads(response.text)
    return response_data
  else:
    print(response.text)
